8    HourlyEmployee::HourlyEmployee()
9    {
10       hourlyRate  = 0;
11       hoursWorked = 0;
12   }
13   
14   HourlyEmployee::HourlyEmployee(string name, int id, double rate, int hours)
15       : Employee(name, id)
16   {
17       hourlyRate  = rate;
18       hoursWorked = hours;
19   }
20   
21   double HourlyEmployee::CalcSalary()
22   {
23       return hourlyRate * hoursWorked;
24   }
25   
26   void HourlyEmployee::DisplayInfo()
27   {
28       Employee::DisplayInfo();
29       cout << "Employee Type: Hourly"           << endl;
30       cout << "Hourly Rate: $"  << hourlyRate   << endl;
31       cout << "Hours Worked: "  << hoursWorked  << endl;
32       cout << "Total Salary: $" << CalcSalary() << endl << endl;
33   }
